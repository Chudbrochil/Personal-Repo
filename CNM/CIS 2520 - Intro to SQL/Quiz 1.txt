SQL 2520 Quiz 1

1.
SELECT TO_CHAR(ADD_MONTHS(SYSDATE, 36),'MM/DD/YYYY') "FUTURE"
FROM DUAL;

2.
SELECT FIRST_NAME, LAST_NAME, JOB_ID,
LOWER(SUBSTR(FIRST_NAME,1,1) || SUBSTR(LAST_NAME,1,5)) "USERID"
FROM EMPLOYEES
ORDER BY LAST_NAME;

3.
SELECT EMPLOYEE_ID, LAST_NAME, DEPARTMENT_ID
FROM EMPLOYEES
WHERE SUBSTR(LAST_NAME,1,1) IN('R','T')
AND DEPARTMENT_ID IN(50,60)
ORDER BY DEPARTMENT_ID DESC, LAST_NAME;

4.
SELECT EMPLOYEE_ID, LAST_NAME, JOB_ID, DEPARTMENT_ID, 
NVL(TO_CHAR(MANAGER_ID), 'YIPEE NO MANAGER !!') "MANAGER_ID"
FROM EMPLOYEES
WHERE DEPARTMENT_ID BETWEEN 60 AND 90
ORDER BY DEPARTMENT_ID, LAST_NAME;

5.
SELECT EMPLOYEE_ID, LAST_NAME, HIRE_DATE, DEPARTMENT_ID, 
ROUND(MONTHS_BETWEEN(SYSDATE, HIRE_DATE),2) "MONTHS_WORKED"
FROM EMPLOYEES
WHERE DEPARTMENT_ID IN(40,50,60)
ORDER BY MONTHS_WORKED DESC;

6.
SELECT LAST_NAME, DEPARTMENT_ID,
TRUNC((MONTHS_BETWEEN(SYSDATE, HIRE_DATE) / 12),2) AS Number_years
FROM EMPLOYEES
ORDER BY DEPARTMENT_ID, Number_years desc;

7.
SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME, JOB_ID,
NEXT_DAY(ADD_MONTHS(SYSDATE, 6),3) AS Next_Meeting
FROM EMPLOYEES
WHERE DEPARTMENT_ID BETWEEN 20 AND 60
ORDER BY DEPARTMENT_ID DESC, LAST_NAME;

8.
SELECT LAST_NAME, SYSDATE, TO_CHAR(BIRTHDATE, 'MM/DD/YYYY') AS BIRTHDAY,
TRUNC((MONTHS_BETWEEN(SYSDATE, BIRTHDATE) / 12),2) AS STAFF_AGE
FROM F_STAFFS;

9.
SELECT EMPLOYEE_ID, LAST_NAME, JOB_ID, DEPARTMENT_ID, SALARY,
(CASE JOB_ID
WHEN 'IT_PROG' THEN SALARY*.3333
WHEN 'FI_ACCOUNT' THEN SALARY*.1333
ELSE 100
END) Bonus_Amount
FROM EMPLOYEES;

10.
SELECT LAST_NAME, DEPARTMENT_ID, SALARY, LPAD(SALARY, 10, '*') "Padded_Salary"
FROM EMPLOYEES;
